}

# 10
result = data.frame(matrix(NA,dim(train_x2)[1]*0.9,2))
names(result) = c("k","mean_re")
for (i in 1:89){
    folds = createFolds(train_y2,k=10)
    re = matrix(NA,10,1)
    for(j in 1:10){  
    traindata_x = train_x2[-folds[[j]],]    
    traindata_y = train_y2[-folds[[j]]]   
    testdata_x = train_x2[folds[[j]],]
    testdata_y = train_y2[folds[[j]]]
    model_f10 = knn(traindata_x,testdata_x,traindata_y, k = i, prob = TRUE)
    re[j] = sum(model_f10 != testdata_y)/length(testdata_y)
    }  
    result[i,1] = i
    result[i,2] = mean(re)#k      
}
