    result[i,1] = i
    result[i,2] = mean(re) #取k折交叉验证结果的均值作为评判模型准确率的结果      
}

# 分离测试集和训练集
set.seed(1)
train2 = sample(1:135,100,F)
train_x2 = data[train2,2:3]
train_y2 = data[train2,7]
test_x2 = data[-train2,2:3]
test_y2 = data[-train2,7]

# 10折交叉验证
result = data.frame(matrix(NA,dim(train_x2)[1]*0.9,2))
names(result) = c("k","mean_re")
for (i in 1:89){
    folds = createFolds(train_y2,k=10)
    re = matrix(NA,10,1)
    for(j in 1:10){  
    traindata_x = train_x2[-folds[[j]],]    
    traindata_y = train_y2[-folds[[j]]]   
    testdata_x = train_x2[folds[[j]],]
    testdata_y = train_y2[folds[[j]]]
    model_f10 = knn(traindata_x,testdata_x,traindata_y, k = i, prob = TRUE)
    re[j] = sum(model_f10 != testdata_y)/length(testdata_y)
    }  
    result[i,1] = i
    result[i,2] = mean(re)#取k折交叉验证结果的均值作为评判模型准确率的结果      
}

# 交叉验证结果(选择k = 8)
ggplot(data = result) + 
  geom_smooth(mapping = aes(x = k, y = mean_re)) + 
  geom_line(mapping = aes(x = k, y = mean_re))
