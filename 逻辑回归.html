# 交叉验证调整阈值
result = data.frame(matrix(NA,51,4))
names(result) = c("thresh","precision","recall","CR")
k = 1
for (i in seq(0,1,0.02)){
    folds = createFolds(train_row,k=10)
    temp_result = data.frame(matrix(NA,10,3))
    for(j in 1:10){  
    traindata = train[-folds[[j]],]      
    testdata = train[folds[[j]],]
    glm.fit=glm(label ~ .,data = traindata,family=binomial)
    glm.probs=predict(glm.fit,newdata= testdata,type="response")
    predict = as.factor(ifelse(glm.probs > i,1,0))   
    temp_result[j,1] = sum(predict ==1 & testdata$label == 1)/sum(predict == 1)
    temp_result[j,2] = sum(predict ==1 & testdata$label == 1)/sum(testdata$label == 1)
    temp_result[j,3] = sum(predict ==1)/length(testdata$label)    
    }  
    result[k,1] = i
    result[k,2] = mean(temp_result[,1])   
    result[k,3] = mean(temp_result[,2])
    result[k,4] = mean(temp_result[,3])
    k = k + 1
}
